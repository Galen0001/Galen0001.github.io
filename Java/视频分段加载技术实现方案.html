Java实现视频上传、FFmpeg分割与分段加载完整方案 ​

下面我将提供一个完整的实现方案，包括Java后端处理上传、FFmpeg分割视频、数据库记录存储以及前端分段加载的实现。

系统架构设计 ​

  1. 上传流程：
    • 前端上传视频文件 → Java后端接收 → FFmpeg分割 → 保存分片文件 → 记录到数据库
  2. 播放流程：
    • 前端请求视频 → 后端查询数据库获取分片信息 → 返回播放列表 → 前端按需加载分片

1. 数据库设计 ​

sql
CREATE TABLE videos (
    id VARCHAR(36) PRIMARY KEY,
    original_name VARCHAR(255),
    title VARCHAR(255),
    duration INT COMMENT '视频时长(秒)',
    status ENUM('UPLOADING', 'PROCESSING', 'READY', 'FAILED'),
    created_at DATETIME,
    updated_at DATETIME
);

CREATE TABLE video_segments (
    id VARCHAR(36) PRIMARY KEY,
    video_id VARCHAR(36),
    segment_index INT,
    file_path VARCHAR(512),
    duration INT COMMENT '分片时长(秒)',
    FOREIGN KEY (video_id) REFERENCES videos(id)
);

@RestController
@RequestMapping("/api/videos")
public class VideoController {
    
    @Autowired
    private VideoService videoService;
    
    @PostMapping("/upload")
    public ResponseEntity<VideoUploadResponse> uploadVideo(
            @RequestParam("file") MultipartFile file,
            @RequestParam(value = "title", required = false) String title) {
        
        VideoUploadResponse response = videoService.processVideoUpload(file, title);
        return ResponseEntity.ok(response);
    }
    
    @GetMapping("/{videoId}/playlist")
    public ResponseEntity<Resource> getVideoPlaylist(@PathVariable String videoId) {
        return videoService.getVideoPlaylist(videoId);
    }
    
    @GetMapping("/segments/{segmentId}")
    public ResponseEntity<Resource> getVideoSegment(@PathVariable String segmentId) {
        return videoService.getVideoSegment(segmentId);
    }
}

视频处理服务 ​

java
@Service
public class VideoService {
    
    @Value("${video.storage.path}")
    private String storagePath;
    
    @Autowired
    private VideoRepository videoRepository;
    
    @Autowired
    private VideoSegmentRepository segmentRepository;
    
    public VideoUploadResponse processVideoUpload(MultipartFile file, String title) {
        // 1. 保存原始文件
        String videoId = UUID.randomUUID().toString();
        String originalName = file.getOriginalFilename();
        
        Video video = new Video();
        video.setId(videoId);
        video.setOriginalName(originalName);
        video.setTitle(title != null ? title : originalName);
        video.setStatus(VideoStatus.UPLOADING);
        video.setCreatedAt(LocalDateTime.now());
        videoRepository.save(video);
        
        try {
            // 2. 保存原始文件
            Path uploadDir = Paths.get(storagePath, videoId);
            Files.createDirectories(uploadDir);
            
            Path originalFilePath = uploadDir.resolve("original.mp4");
            file.transferTo(originalFilePath.toFile());
            
            // 3. 异步处理视频分割
            CompletableFuture.runAsync(() -> processVideoSegmentation(videoId));
            
            return new VideoUploadResponse(videoId, "视频上传成功，正在处理");
        } catch (IOException e) {
            video.setStatus(VideoStatus.FAILED);
            videoRepository.save(video);
            throw new RuntimeException("视频上传失败", e);
        }
    }
    
    private void processVideoSegmentation(String videoId) {
        Video video = videoRepository.findById(videoId)
                .orElseThrow(() -> new RuntimeException("视频不存在"));
        
        try {
            video.setStatus(VideoStatus.PROCESSING);
            videoRepository.save(video);
            
            Path videoDir = Paths.get(storagePath, videoId);
            Path originalFile = videoDir.resolve("original.mp4");
            Path outputDir = videoDir.resolve("segments");
            Files.createDirectories(outputDir);
            
            // 使用FFmpeg分割视频
            String m3u8Path = outputDir.resolve("playlist.m3u8").toString();
            
            String command = String.format(
                "ffmpeg -i %s -c:v copy -c:a copy -f hls -hls_time 10 -hls_list_size 0 %s",
                originalFile.toString(), m3u8Path);
            
            Process process = Runtime.getRuntime().exec(command);
            int exitCode = process.waitFor();
            
            if (exitCode != 0) {
                throw new RuntimeException("视频处理失败");
            }
            
            // 解析M3U8文件，获取分片信息并存入数据库
            List<VideoSegment> segments = parseM3u8AndSaveSegments(videoId, outputDir);
            
            // 更新视频状态
            video.setStatus(VideoStatus.READY);
            video.setDuration(calculateTotalDuration(segments));
            videoRepository.save(video);
            
        } catch (Exception e) {
            video.setStatus(VideoStatus.FAILED);
            videoRepository.save(video);
            throw new RuntimeException("视频处理失败", e);
        }
    }
    
    private List<VideoSegment> parseM3u8AndSaveSegments(String videoId, Path outputDir) throws IOException {
        List<VideoSegment> segments = new ArrayList<>();
        Path m3u8File = outputDir.resolve("playlist.m3u8");
        
        List<String> lines = Files.readAllLines(m3u8File);
        Pattern pattern = Pattern.compile("segment(\\d+)\\.ts");
        
        int index = 0;
        for (String line : lines) {
            if (line.endsWith(".ts")) {
                Matcher matcher = pattern.matcher(line);
                if (matcher.find()) {
                    VideoSegment segment = new VideoSegment();
                    segment.setId(UUID.randomUUID().toString());
                    segment.setVideoId(videoId);
                    segment.setSegmentIndex(index++);
                    segment.setFilePath(outputDir.resolve(line).toString());
                    segment.setDuration(10); // 假设每个分片10秒
                    
                    segmentRepository.save(segment);
                    segments.add(segment);
                }
            }
        }
        
        return segments;
    }
    
    public ResponseEntity<Resource> getVideoPlaylist(String videoId) {
        Video video = videoRepository.findById(videoId)
                .orElseThrow(() -> new RuntimeException("视频不存在"));
        
        if (video.getStatus() != VideoStatus.READY) {
            throw new RuntimeException("视频尚未处理完成");
        }
        
        Path playlistPath = Paths.get(storagePath, videoId, "segments", "playlist.m3u8");
        Resource resource = new FileSystemResource(playlistPath);
        
        return ResponseEntity.ok()
                .header("Content-Type", "application/x-mpegURL")
                .body(resource);
    }
    
    public ResponseEntity<Resource> getVideoSegment(String segmentId) {
        VideoSegment segment = segmentRepository.findById(segmentId)
                .orElseThrow(() -> new RuntimeException("分片不存在"));
        
        Resource resource = new FileSystemResource(segment.getFilePath());
        
        return ResponseEntity.ok()
                .header("Content-Type", "video/MP2T")
                .body(resource);
    }
}

HTML播放器页面 ​

<!DOCTYPE html>
<html>
<head>
    <title>视频播放器</title>
    <script src="https://cdn.jsdelivr.net/npm/hls.js@latest"></script>
</head>
<body>
    <video id="video" controls width="640"></video>
    
    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const video = document.getElementById('video');
            const videoId = '从URL获取的视频ID';
            
            if (Hls.isSupported()) {
                const hls = new Hls();
                hls.loadSource(`/api/videos/${videoId}/playlist`);
                hls.attachMedia(video);
                hls.on(Hls.Events.MANIFEST_PARSED, function() {
                    video.play();
                });
            } else if (video.canPlayType('application/vnd.apple.mpegurl')) {
                // 原生支持HLS的浏览器(如Safari)
                video.src = `/api/videos/${videoId}/playlist`;
                video.addEventListener('loadedmetadata', function() {
                    video.play();
                });
            }
        });
    </script>
</body>
</html>

文件上传组件(React示例) ​

jsx
import React, { useState } from 'react';
import axios from 'axios';

function VideoUpload() {
    const [file, setFile] = useState(null);
    const [title, setTitle] = useState('');
    const [uploadStatus, setUploadStatus] = useState('');
    const [videoId, setVideoId] = useState(null);
    
    const handleUpload = async () => {
        if (!file) return;
        
        const formData = new FormData();
        formData.append('file', file);
        if (title) formData.append('title', title);
        
        try {
            setUploadStatus('上传中...');
            const response = await axios.post('/api/videos/upload', formData, {
                headers: { 'Content-Type': 'multipart/form-data' }
            });
            
            setVideoId(response.data.videoId);
            setUploadStatus('上传成功，视频处理中...');
            
            // 可以轮询检查视频处理状态
            checkVideoStatus(response.data.videoId);
        } catch (error) {
            setUploadStatus('上传失败');
            console.error(error);
        }
    };
    
    const checkVideoStatus = async (id) => {
        const interval = setInterval(async () => {
            try {
                const response = await axios.get(`/api/videos/${id}/status`);
                if (response.data.status === 'READY') {
                    clearInterval(interval);
                    setUploadStatus('视频处理完成，可以播放');
                } else if (response.data.status === 'FAILED') {
                    clearInterval(interval);
                    setUploadStatus('视频处理失败');
                }
            } catch (error) {
                clearInterval(interval);
                setUploadStatus('状态检查失败');
            }
        }, 3000);
    };
    
    return (
        <div>
            <input type="file" accept="video/*" onChange={(e) => setFile(e.target.files[0])} />
            <input type="text" placeholder="视频标题" value={title} onChange={(e) => setTitle(e.target.value)} />
            <button onClick={handleUpload}>上传</button>
            <p>{uploadStatus}</p>
            {videoId && (
                <a href={`/player.html?videoId=${videoId}`}>观看视频</a>
            )}
        </div>
    );
}

export default VideoUpload;

4. 部署与优化建议 ​

  1. FFmpeg安装：

    • 确保服务器安装了FFmpeg并添加到系统PATH
    • 可以使用静态构建版本：https://johnvansickle.com/ffmpeg/

  2. 存储优化：

    yaml
    # application.yml配置示例
video:
  storage:
    path: /var/video-storage
    max-file-size: 2GB

  3. 性能优化：

    • 使用线程池处理视频分割任务
    • 对大文件使用分块上传
    • 考虑使用消息队列(MQ)处理视频转码任务

  4. 安全考虑：

    • 限制上传文件类型
    • 设置文件大小限制
    • 对上传文件进行病毒扫描

  5. 扩展性：

    • 对于高负载场景，考虑使用分布式转码
    • 使用云存储服务(S3等)替代本地存储

这个完整方案实现了从视频上传、处理到分段播放的整个流程，可以根据实际需求进行调整和扩展。
